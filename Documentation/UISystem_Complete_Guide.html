UIService - Main Service Controller

The central hub that orchestrates all UI operations through a component-based architecture.

UIService.cs - Core Implementation 
[EngineService(ServiceCategory.Core, ServicePriority.High, 
    Description = "Component-based UI service with improved architecture")]
[ServiceConfiguration(typeof(UIServiceConfiguration))]
public class UIService : IUIService, IUIServiceContext
{
    private readonly UIServiceConfiguration _config;
    private readonly IResourceService _resourceService;
    private readonly UIScreenRegistry _screenRegistry;
    private readonly Dictionary<Type, IUIComponent> _components;
    
    // Component-based delegation for better maintainability
    public async UniTask ShowAsync(UIScreenType screenType, 
        UIScreenContext context = null, 
        UIDisplayConfig displayConfig = null, 
        CancellationToken cancellationToken = default)
    {
        var screenAsset = _screenRegistry.GetAsset(screenType);
        var navManager = GetComponent<UINavigationManager>();
        await navManager.ShowScreenAsync(screenAsset, context, displayConfig, cancellationToken);
    }
    
    public IUINavigationBuilder Navigate() => new UINavigationBuilder(this);
}
Key Design Decision: The UIService delegates all operations to specialized component managers, following single responsibility principle and improving maintainability.

UITransitionManager - Animation System ‚ñº

DOTween-powered transition system with caching and performance optimization.

Transition System Architecture 
public class UITransitionManagerComponent : IUIComponent
{
    private readonly Dictionary<TransitionType, IUITransition> _transitions;
    
    public async UniTask ExecuteTransitionAsync(UIScreen screen, 
        TransitionType type, float duration, CancellationToken cancellationToken)
    {
        var transition = GetOrCreateTransition(type);
        await transition.ExecuteAsync(screen, duration, cancellationToken);
    }
    
    // Supported transition types: Fade, Slide, Scale, Push
    private IUITransition GetOrCreateTransition(TransitionType type)
    {
        if (!_transitions.ContainsKey(type))
        {
            _transitions[type] = type switch
            {
                TransitionType.Fade => new FadeTransition(),
                TransitionType.SlideLeft => new SlideTransition(SlideDirection.Left),
                TransitionType.ScaleUp => new ScaleTransition(ScaleMode.Up),
                TransitionType.PushLeft => new PushTransition(PushDirection.Left),
                _ => new FadeTransition()
            };
        }
        return _transitions[type];
    }
}

Caching Strategy 
public class UIScreenCacheManager : IUIComponent, IMemoryPressureResponder
{
    private readonly LRUCache<UIScreenType, UIScreen> _screenCache;
    
    public void RespondToMemoryPressure(float pressureLevel)
    {
        if (pressureLevel > 0.8f)
        {
            // Aggressive cleanup - remove 50% of cached screens
            _screenCache.RemoveOldest(_screenCache.Count / 2);
        }
        else if (pressureLevel > 0.6f)
        {
            // Moderate cleanup - remove 25% of cached screens  
            _screenCache.RemoveOldest(_screenCache.Count / 4);
        }
    }
}

UIScreen Base Class - Simple but Powerful 
/// <summary>
/// Base class for all UI screens with simple lifecycle
/// </summary>
[RequireComponent(typeof(Canvas))]
public abstract class UIScreen : MonoBehaviour
{
    [Header("Screen Configuration")]
    [SerializeField] private UIScreenAsset _screenAsset;
    [SerializeField] private UIDisplayConfig _displayConfig;
    
    public UIScreenAsset ScreenAsset { get; internal set; }
    public UIDisplayConfig DisplayConfig { get; internal set; }
    public bool IsActive { get; private set; }
    public Canvas Canvas { get; private set; }
    
    // Simple lifecycle - only 4 methods to override
    protected virtual void OnShow(UIScreenContext context) { }
    protected virtual void OnHide() { }
    protected virtual void OnDestroy() { }
    public virtual bool OnBackPressed() => true; // Return false to consume back press
    
    // Internal lifecycle management
    internal async UniTask ShowAsync(UIScreenContext context)
    {
        IsActive = true;
        gameObject.SetActive(true);
        OnShow(context);
    }
    
    internal async UniTask HideAsync()
    {
        OnHide();
        IsActive = false;
        gameObject.SetActive(false);
    }
}

The Navigation Builder provides a fluent API for complex navigation scenarios with method chaining.

UINavigationBuilder - Fluent Interface 
public interface IUINavigationBuilder
{
    #region Target Selection
    IUINavigationBuilder To(UIScreenType screenType);
    IUINavigationBuilder ToAsset(UIScreenAsset screenAsset);
    #endregion
    
    #region Data & Context
    IUINavigationBuilder WithData(object data);
    IUINavigationBuilder WithContext(UIScreenContext context);
    #endregion
    
    #region Display Modes
    IUINavigationBuilder AsModal();
    IUINavigationBuilder AsOverlay();
    IUINavigationBuilder WithDisplayConfig(UIDisplayConfig config);
    #endregion
    
    #region Transitions
    IUINavigationBuilder WithTransition(TransitionType transitionType);
    IUINavigationBuilder WithInTransition(TransitionType transitionType);
    IUINavigationBuilder WithOutTransition(TransitionType transitionType);
    IUINavigationBuilder WithTransitionDuration(float duration);
    #endregion
    
    #region Navigation Actions
    IUINavigationBuilder Push(); // Default - adds to stack
    IUINavigationBuilder Replace(); // Replaces current screen

    #endregion
    
    #region Execution

    #endregion
}

protected override void OnHide()
{
    // Called when screen is being hidden
    // - GameObject is still active during this call
    // - Out-transition will be applied after this method
    // - Use this for cleanup and state saving
    
    // Example implementation:

    if (tempTexture != null)
    {
        DestroyImmediate(tempTexture);
        tempTexture = null;
    }
}

public override bool OnBackPressed()
{
    // Handle back button (ESC key, Android back, etc.)
    // Return true to allow default back navigation
    // Return false to consume the back press
    
    if (hasUnsavedChanges)
    {
        // Show confirmation dialog
        ShowUnsavedChangesDialog();
        return false; // Consume back press, don't navigate

public class GameController : MonoBehaviour
{
    private IUIService uiService;

    {
        // Get UI service
        uiService = Engine.GetService<IUIService>();
        
        // Show initial screen with player data

    private void Update()
    {
        // Handle global back button (ESC, Android back)
        if (Input.GetKeyDown(KeyCode.Escape))
        {
            HandleBackPressed();
        }
    }
    
    private async void HandleBackPressed()
    {
        var currentScreen = uiService.GetActiveScreen();
        if (currentScreen != null && !currentScreen.OnBackPressed())
        {
            // Screen consumed the back press
            return;
        }
        
        // Default back navigation
        if (uiService.GetStackDepth() > 1)
        {
            await uiService.PopAsync();
        }
        else
        {
            // At root, show quit confirmation
            await ShowQuitConfirmation();
        }
    }
}

public class ConfirmDialogScreen : UIScreen
{
    [Header("Dialog UI")]
    [SerializeField] private Text titleText;
    [SerializeField] private Text messageText;
    [SerializeField] private Button confirmButton;
    [SerializeField] private Button cancelButton;
    [SerializeField] private Button closeButton;
    
    public event System.Action<bool> DialogResult;
    
    protected override void OnShow(UIScreenContext context)
    {
        if (context?.Data is ConfirmDialogData dialogData)
        {
            titleText.text = dialogData.Title;
            messageText.text = dialogData.Message;
            confirmButton.GetComponentInChildren<Text>().text = dialogData.ConfirmText ?? "OK";
            cancelButton.GetComponentInChildren<Text>().text = dialogData.CancelText ?? "Cancel";
        }
        
        confirmButton.onClick.AddListener(() => CloseDialog(true));
        cancelButton.onClick.AddListener(() => CloseDialog(false));
        closeButton.onClick.AddListener(() => CloseDialog(false));
    }
    
    public override bool OnBackPressed()
    {
        // Treat back press as cancel
        CloseDialog(false);
        return false; // Consume back press
    }
    
    private async void CloseDialog(bool confirmed)
    {
        DialogResult?.Invoke(confirmed);
        
        var uiService = Engine.GetService<IUIService>();
        await uiService.HideAsync(UIScreenType.ConfirmDialog);
    }
}

    public static async UniTask<bool> ShowConfirmationAsync(string title, string message, 
        string confirmText = "OK", string cancelText = "Cancel")
    {
        var uiService = Engine.GetService<IUIService>();
        var taskCompletionSource = new UniTaskCompletionSource<bool>();
        
        var dialogData = new ConfirmDialogData
        {
            Title = title,
            Message = message,
            ConfirmText = confirmText,
            CancelText = cancelText
        };
        
        var context = new UIScreenContext { Data = dialogData };
        
        await uiService.Navigate()
            .To(UIScreenType.ConfirmDialog)
            .WithContext(context)
            .AsModal()
            .WithTransition(TransitionType.ScaleUp)
            .ExecuteAsync();
            
        // Subscribe to dialog result
        var screen = uiService.GetActiveScreen() as ConfirmDialogScreen;
        screen.DialogResult += result =>
        {
            taskCompletionSource.TrySetResult(result);
        };
        
        return await taskCompletionSource.Task;
    }
    
    public static async UniTask ShowErrorAsync(string title, string message)
    {
        await ShowConfirmationAsync(title, message, "OK", null);
    }
    
    public static async UniTask ShowInfoAsync(string title, string message)
    {
        await ShowConfirmationAsync(title, message, "OK", null);
    }
}

                scoreText.text = $"Score: {gameplayData.Score:N0}";
                timeText.text = $"Time: {FormatTime(gameplayData.ElapsedTime)}";

            yield return new WaitForSeconds(0.1f); // 10 FPS update rate
        }
    }

        await uiService.Navigate()
            .To(UIScreenType.PauseMenu)

public class SettingsScreen : UIScreen
{
    [Header("Settings UI")]
    [SerializeField] private Slider masterVolumeSlider;
    [SerializeField] private Slider musicVolumeSlider;
    [SerializeField] private Slider sfxVolumeSlider;
    [SerializeField] private Toggle fullscreenToggle;
    [SerializeField] private Dropdown resolutionDropdown;
    [SerializeField] private Button saveButton;
    [SerializeField] private Button cancelButton;
    [SerializeField] private Button resetButton;
    
    private GameSettings originalSettings;
    private GameSettings currentSettings;
    private ISaveLoadService saveLoadService;
    
    protected override async void OnShow(UIScreenContext context)
    {
        saveLoadService = Engine.GetService<ISaveLoadService>();
        
        // Load current settings
        await LoadCurrentSettings();
        
        // Setup UI
        SetupUI();
        
        // Subscribe to events
        saveButton.onClick.AddListener(OnSaveClicked);
        cancelButton.onClick.AddListener(OnCancelClicked);
        resetButton.onClick.AddListener(OnResetClicked);
        
        // Subscribe to value changes
        masterVolumeSlider.onValueChanged.AddListener(OnMasterVolumeChanged);
        musicVolumeSlider.onValueChanged.AddListener(OnMusicVolumeChanged);
        sfxVolumeSlider.onValueChanged.AddListener(OnSFXVolumeChanged);
        fullscreenToggle.onValueChanged.AddListener(OnFullscreenChanged);
        resolutionDropdown.onValueChanged.AddListener(OnResolutionChanged);
    }
    
    private async UniTask LoadCurrentSettings()
    {
        var result = await saveLoadService.LoadAsync<GameSettings>("game_settings");
        if (result.IsSuccess)
        {
            originalSettings = result.Data;
            currentSettings = result.Data.Clone();
        }
        else
        {
            // Use default settings
            originalSettings = GameSettings.Default;
            currentSettings = GameSettings.Default.Clone();
        }
    }
    
    private void SetupUI()
    {
        masterVolumeSlider.value = currentSettings.MasterVolume;
        musicVolumeSlider.value = currentSettings.MusicVolume;
        sfxVolumeSlider.value = currentSettings.SFXVolume;
        fullscreenToggle.isOn = currentSettings.Fullscreen;
        
        // Populate resolution dropdown
        resolutionDropdown.options.Clear();
        for (int i = 0; i < Screen.resolutions.Length; i++)
        {
            var res = Screen.resolutions[i];
            resolutionDropdown.options.Add(new Dropdown.OptionData($"{res.width}x{res.height}"));
            
            if (res.width == currentSettings.ResolutionWidth && 
                res.height == currentSettings.ResolutionHeight)
            {
                resolutionDropdown.value = i;
            }
        }
    }
    
    private async void OnSaveClicked()
    {
        // Apply settings to game
        ApplySettingsToGame(currentSettings);
        
        // Save to disk
        await saveLoadService.SaveAsync("game_settings", currentSettings);
        
        // Close settings screen
        var uiService = Engine.GetService<IUIService>();
        await uiService.HideAsync(UIScreenType.Settings);
    }
    
    private async void OnCancelClicked()
    {
        // Revert to original settings
        ApplySettingsToGame(originalSettings);
        
        // Close without saving
        var uiService = Engine.GetService<IUIService>();
        await uiService.HideAsync(UIScreenType.Settings);
    }
    
    private async void OnResetClicked()
    {
        // Show confirmation
        var confirmed = await DialogHelper.ShowConfirmationAsync(
            "Reset Settings", 
            "This will reset all settings to default values. Continue?");
            
        if (confirmed)
        {
            currentSettings = GameSettings.Default.Clone();
            SetupUI();
            ApplySettingsToGame(currentSettings);
        }
    }
    
    public override bool OnBackPressed()
    {
        // Check for unsaved changes
        if (!currentSettings.Equals(originalSettings))
        {
            ShowUnsavedChangesDialog();
            return false; // Consume back press
        }
        
        return true; // Allow normal back navigation
    }
    
    private async void ShowUnsavedChangesDialog()
    {
        var result = await DialogHelper.ShowConfirmationAsync(
            "Unsaved Changes", 
            "You have unsaved changes. Save before closing?",
            "Save", "Discard");
            
        if (result)
        {
            OnSaveClicked();
        }
        else
        {
            OnCancelClicked();
        }
    }
}

public class UIScreenCacheManager : IUIComponent, IMemoryPressureResponder
{
    private readonly LRUCache<UIScreenType, UIScreen> _cache;
    private int _cacheHits = 0;
    private int _cacheMisses = 0;
    
    public double HitRatio => _cacheHits + _cacheMisses > 0 ? 
        (double)_cacheHits / (_cacheHits + _cacheMisses) : 0.0;
    
    public void RespondToMemoryPressure(float pressureLevel)
    {
        var itemsToRemove = pressureLevel switch
        {
            > 0.8f => _cache.Count / 2,    // Aggressive cleanup
            > 0.6f => _cache.Count / 4,    // Moderate cleanup  
            > 0.4f => _cache.Count / 8,    // Light cleanup
            _ => 0
        };
        
        if (itemsToRemove > 0)
        {
            _cache.RemoveOldest(itemsToRemove);
            Debug.Log($"[UIScreenCache] Cleaned {itemsToRemove} screens due to memory pressure");
        }
    }
}

üé≠ Object Pooling

public class UIInstancePoolManager : IUIComponent
{
    private readonly Dictionary<UIScreenType, ObjectPool<UIScreen>> _pools;
    
    public (int available, int active, int totalCreated, double efficiency) 
        GetStats(UIScreenType screenType)
    {
        if (_pools.TryGetValue(screenType, out var pool))
        {
            return (pool.Available, pool.Active, pool.TotalCreated, pool.Efficiency);
        }
        return (0, 0, 0, 0.0);
    }
    
    public UIScreen GetOrCreate(UIScreenType screenType)
    {
        if (!_pools.TryGetValue(screenType, out var pool))
        {
            pool = CreatePool(screenType);
            _pools[screenType] = pool;
        }
        
        return pool.Get();
    }
}
