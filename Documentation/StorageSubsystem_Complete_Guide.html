Overview

The Storage Subsystem is a comprehensive save/load solution for the Sinkii09 Engine, providing a flexible, extensible, and platform-agnostic way to persist game data. It supports multiple storage backends, automatic provider discovery, health monitoring, and advanced features like compression, encryption, and cloud synchronization.

Core Components

SaveLoadService

The main service that provides the high-level API for save/load operations. It handles serialization, compression, encryption, and coordinates with storage providers.

Key Responsibilities:

  • Data serialization and deserialization
  • Compression and decompression
  • Encryption and decryption (optional)
  • Provider initialization and management
  • Event notifications
  • Statistics tracking
// SaveLoadService API
public interface ISaveLoadService : IEngineService
{
    // Core Operations
    UniTask<SaveResult> SaveAsync(string saveId, SaveData data, CancellationToken cancellationToken = default);
    UniTask<LoadResult<T>> LoadAsync<T>(string saveId, CancellationToken cancellationToken = default) where T : SaveData;
    UniTask<bool> ExistsAsync(string saveId, CancellationToken cancellationToken = default);
    UniTask<bool> DeleteAsync(string saveId, CancellationToken cancellationToken = default);
    
    // Metadata Operations
    UniTask<IReadOnlyList<SaveMetadata>> GetAllSavesAsync(CancellationToken cancellationToken = default);
    UniTask<SaveMetadata> GetSaveMetadataAsync(string saveId, CancellationToken cancellationToken = default);
    
    // Backup Operations
    UniTask<bool> CreateBackupAsync(string saveId, CancellationToken cancellationToken = default);
    UniTask<bool> RestoreBackupAsync(string saveId, string backupId, CancellationToken cancellationToken = default);
    UniTask<IReadOnlyList<string>> GetBackupsAsync(string saveId, CancellationToken cancellationToken = default);
    
    // Utility Operations
    UniTask<bool> ValidateSaveAsync(string saveId, CancellationToken cancellationToken = default);
    UniTask<SaveResult> AutoSaveAsync(SaveData data, CancellationToken cancellationToken = default);
    UniTask<LoadResult<T>> LoadLatestAsync<T>(CancellationToken cancellationToken = default) where T : SaveData;
    
    // Statistics
    SaveLoadServiceStatistics GetStatistics();
    void ResetStatistics();
}

StorageManager

Manages multiple storage providers, handles routing, failover, and aggregates operations across providers.

Features:

  • Provider registration and lifecycle management
  • Automatic failover between providers
  • Health monitoring integration
  • Aggregated statistics
  • Load balancing (future)
Tip: StorageManager automatically registers a LocalFileStorage provider if no providers are configured, ensuring the system always has at least one functional storage backend.

Usage Guide

Basic Save/Load Operations

// Define your save data class
[Serializable]
public class GameSaveData : SaveData
{
    public string PlayerName { get; set; }
    public int Level { get; set; }
    public float PlayTime { get; set; }
    public List<string> UnlockedItems { get; set; }
}

// Save data
public async UniTask SaveGameAsync()
{
    var saveData = new GameSaveData
    {
        PlayerName = "Player1",
        Level = 10,
        PlayTime = 3600f,
        UnlockedItems = new List<string> { "sword", "shield", "potion" }
    };
    
    var saveLoadService = Engine.GetService<ISaveLoadService>();
    var result = await saveLoadService.SaveAsync("save_slot_1", saveData);
    
    if (result.Success)
    {
        Debug.Log($"Game saved successfully in {result.Duration.TotalMilliseconds}ms");
        Debug.Log($"Compressed size: {result.CompressedSize} bytes");
    }
    else
    {
        Debug.LogError($"Save failed: {result.ErrorMessage}");
    }
}
// Load data
public async UniTask LoadGameAsync()
{
    var saveLoadService = Engine.GetService<ISaveLoadService>();
    var result = await saveLoadService.LoadAsync<GameSaveData>("save_slot_1");
    
    if (result.Success)
    {
        var saveData = result.Data;
        Debug.Log($"Loaded game for player: {saveData.PlayerName}");
        Debug.Log($"Level: {saveData.Level}, Play time: {saveData.PlayTime}s");
        
        // Apply loaded data to game
        ApplyLoadedData(saveData);
    }
    else if (result.NotFound)
    {
        Debug.LogWarning("Save file not found");
    }
    else
    {
        Debug.LogError($"Load failed: {result.ErrorMessage}");
    }
}
// Check if save exists
public async UniTask<bool> HasSaveGameAsync(string saveId)
{
    var saveLoadService = Engine.GetService<ISaveLoadService>();
    return await saveLoadService.ExistsAsync(saveId);
}

// List all saves
public async UniTask ListSavesAsync()
{
    var saveLoadService = Engine.GetService<ISaveLoadService>();
    var saves = await saveLoadService.GetAllSavesAsync();
    
    foreach (var save in saves)
    {
        Debug.Log($"Save: {save.SaveId}");
        Debug.Log($"  Created: {save.CreatedAt}");
        Debug.Log($"  Modified: {save.ModifiedAt}");
        Debug.Log($"  Size: {save.FileSize} bytes");
    }
}
// Delete a save
public async UniTask DeleteSaveAsync(string saveId)
{
    var saveLoadService = Engine.GetService<ISaveLoadService>();
    
    // Optional: Create backup before deleting
    await saveLoadService.CreateBackupAsync(saveId);
    
    bool deleted = await saveLoadService.DeleteAsync(saveId);
    
    if (deleted)
    {
        Debug.Log($"Save '{saveId}' deleted successfully");
    }
    else
    {
        Debug.LogError($"Failed to delete save '{saveId}'");
    }
}

Cloud Synchronization

// Using cloud storage providers
public class CloudSyncManager
{
    private GoogleDriveStorage _cloudStorage;
    
    public async UniTask SetupCloudSyncAsync()
    {
        // Get cloud storage provider
        var storageManager = Engine.GetService<ISaveLoadService>()
            .GetStorageManager();
        
        _cloudStorage = storageManager.GetProvider<GoogleDriveStorage>();
        
        // Authenticate
        var credentials = new CloudCredentials
        {
            AccessToken = PlayerPrefs.GetString("GoogleAccessToken"),
            RefreshToken = PlayerPrefs.GetString("GoogleRefreshToken")
        };
        
        var authResult = await _cloudStorage.AuthenticateAsync(credentials);
        
        if (authResult.Success)
        {
            // Start synchronization
            var syncOptions = new CloudSyncOptions
            {
                UploadLocalChanges = true,
                DownloadCloudChanges = true,
                DefaultConflictResolution = CloudConflictResolution.KeepNewer
            };
            
            var syncResult = await _cloudStorage.SyncAsync(syncOptions);
            Debug.Log($"Synced {syncResult.FilesUploaded} uploads, {syncResult.FilesDownloaded} downloads");
        }
    }
}

Storage Providers

LocalFileStorage

The default storage provider that saves data to the local file system. It provides robust file operations with atomic writes and automatic backup support.

Features:

  • Atomic file operations (write to temp, then move)
  • Automatic directory creation
  • Metadata files for save information
  • Built-in backup/restore functionality
  • Cross-platform file path handling

File Structure:

├── save_slot_2.sav

└── Backups/

    └── save_slot_1_backup_20250119_143022.meta
Platform Note: On mobile platforms, files are saved to Application.persistentDataPath to ensure they persist between app updates.

Creating a Custom Storage Provider

Follow these steps to create your own storage provider:

Step 1: Create the Provider Class

using System;
using Cysharp.Threading.Tasks;
using UnityEngine;

namespace Sinkii09.Engine.Services
{
    [StorageProvider(StorageProviderType.Custom,
        Description = "My custom storage provider",
        SupportedPlatforms = SupportedPlatform.All)]
    public class MyCustomStorage : IStorageProvider
    {
        public StorageProviderType ProviderType => StorageProviderType.Custom;
        public StorageHealthStatus HealthStatus { get; private set; }
        
        private bool _isInitialized;
        private string _storagePath;
        
        public async UniTask<StorageInitializationResult> InitializeAsync(
            StorageProviderConfiguration configuration, 
            CancellationToken cancellationToken = default)
        {
            try
            {
                // Initialize your storage backend
                _storagePath = configuration.BasePath;
                
                // Perform any necessary setup
                await SetupStorageBackendAsync(cancellationToken);
                
                _isInitialized = true;
                HealthStatus = StorageHealthStatus.Healthy;
                
                return StorageInitializationResult.CreateSuccess(TimeSpan.Zero);
            }
            catch (Exception ex)
            {
                HealthStatus = StorageHealthStatus.Unhealthy;
                return StorageInitializationResult.CreateFailure(ex.Message, ex);
            }
        }
        
        public async UniTask<StorageOperationResult> SaveAsync(
            string saveId, 
            byte[] data, 
            SaveMetadata metadata = null, 
            CancellationToken cancellationToken = default)
        {
            ValidateInitialization();
            
            try
            {
                // Implement your save logic
                await WriteDataToBackendAsync(saveId, data, cancellationToken);
                
                return StorageOperationResult.CreateSuccess(
                    TimeSpan.FromMilliseconds(100), 
                    data.Length);
            }
            catch (Exception ex)
            {
                return StorageOperationResult.CreateFailure(ex.Message, ex);
            }
        }
        
        public async UniTask<StorageLoadResult> LoadAsync(
            string saveId, 
            CancellationToken cancellationToken = default)
        {
            ValidateInitialization();
            
            try
            {
                // Implement your load logic
                var data = await ReadDataFromBackendAsync(saveId, cancellationToken);
                var metadata = await ReadMetadataAsync(saveId, cancellationToken);
                
                return StorageLoadResult.CreateSuccess(
                    data, 
                    metadata, 
                    TimeSpan.FromMilliseconds(100));
            }
            catch (Exception ex)
            {
                return StorageLoadResult.CreateFailure(ex.Message, ex);
            }
        }
        
        // Implement other required methods...
        
        private void ValidateInitialization()
        {
            if (!_isInitialized)
                throw new InvalidOperationException("Storage provider not initialized");
        }
    }
}

Best Practices

Save Data Design

✅ DO:

  • Keep save data classes simple and serializable
  • Version your save data for backward compatibility
  • Use primitive types and simple collections when possible
  • Separate game progress from settings data
  • Implement data validation in your save classes

❌ DON'T:

  • Save references to Unity objects (use IDs instead)
  • Include transient or calculated data
  • Use complex nested structures unnecessarily
  • Save sensitive information without encryption
  • Assume save/load operations are instant

Performance Optimization

// 1. Use async operations properly
public async UniTask SaveAllGameDataAsync()
{
    var tasks = new List<UniTask<SaveResult>>();
    
    // Save different data types in parallel
    tasks.Add(_saveLoadService.SaveAsync("player", _playerData));
    tasks.Add(_saveLoadService.SaveAsync("settings", _settingsData));
    tasks.Add(_saveLoadService.SaveAsync("progress", _progressData));
    
    var results = await UniTask.WhenAll(tasks);
    
    // Check all results
    foreach (var result in results)
    {
        if (!result.Success)
        {
            Debug.LogError($"Save failed: {result.ErrorMessage}");
        }
    }
}

// 2. Implement save data pooling for frequently saved data
public class SaveDataPool<T> where T : SaveData, new()
{
    private readonly Stack<T> _pool = new Stack<T>();
    
    public T Get()
    {
        return _pool.Count > 0 ? _pool.Pop() : new T();
    }
    
    public void Return(T item)
    {
        // Reset item state
        item.ModifiedAt = DateTime.UtcNow;
        _pool.Push(item);
    }
}

Error Handling

public class SaveGameManager
{
    private readonly ISaveLoadService _saveLoadService;
    private readonly int _maxRetries = 3;
    
    public async UniTask<bool> SaveWithRetryAsync(string saveId, SaveData data)
    {
        for (int attempt = 0; attempt < _maxRetries; attempt++)
        {
            try
            {
                var result = await _saveLoadService.SaveAsync(saveId, data);
                
                if (result.Success)
                {
                    return true;
                }
                
                // Log error and retry
                Debug.LogWarning($"Save attempt {attempt + 1} failed: {result.ErrorMessage}");
                
                // Wait before retry with exponential backoff
                await UniTask.Delay(TimeSpan.FromSeconds(Math.Pow(2, attempt)));
            }
            catch (Exception ex)
            {
                Debug.LogError($"Save exception on attempt {attempt + 1}: {ex.Message}");
                
                if (attempt == _maxRetries - 1)
                {
                    // Final attempt failed, handle critical error
                    HandleCriticalSaveError(ex);
                    return false;
                }
            }
        }
        
        return false;
    }
    
    private void HandleCriticalSaveError(Exception ex)
    {
        // Show user notification
        // Log to analytics
        // Attempt emergency backup
        // Consider switching to alternative storage provider
    }
}

Security Considerations

  • Encryption: Enable encryption for sensitive data like player progress or in-app purchases
  • Integrity Checks: Use checksums to detect save file tampering
  • Backup Strategy: Maintain multiple backups and validate before overwriting
  • Cloud Security: Use secure authentication and never store credentials in code
  • Data Privacy: Follow platform guidelines for data collection and storage

        Directory.Delete("TestSaves", true);
    }
